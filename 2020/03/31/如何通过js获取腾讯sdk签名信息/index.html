如何通过js获取腾讯sdk签名信息

上一节完成录音文件分段，就想着能不能调用第三方接口来实现实时语音转文字，，通过百度搜素后发现好像腾讯体验免费，具体免不免费在那也不知道，咱也不敢问，毕竟注册的时候还让绑定了银行卡号。。。。

  1. 注册

官方文档

首先根据新手指引，完成应用的创建，得到appid和key来用来获取签名。

下面是创建成功的样子：

  1. 获取签名

    参考文献：javascript实现腾讯AI开放平台，调用API时的接口鉴权，生成sign合法签名
    JavaScript-MD5
    javascript实现PHP字典排序ksort

    通过以上途径获取到MD5，PHP_ksort，以及自己即将要书写的文件ret_sign，把他们放在了一个文件夹里，方便以后的复制粘贴，哈哈哈

  1. 代码实现：
     const MD = require("./jsMD5/md5.js");
   const ksort = require("./PHP_ksort.js");

     //module.exports=
      (function createSign(vm, data) {
     let time_str = randomString();
    console.log(time_str);
    let newData = {
    'appid_id': '2131664075',
    'time_stamp': time_str.timestamp,
    'nonce_str': time_str.randomStr,
    'key1': '腾讯AI开放平台',
    'key2': '实例仅供参考'
    }

    // return new Promise((resolve)=>{
    //字典排序
    let aa = ksort(vm, newData)
    console.log(aa);

    let bb = ''

    // 拼按URL键值对
    Object.keys(aa).forEach((key, i) => {
       if (aa[key] !== '') {
        bb += key + '=' + encodeURIComponent(aa[key]) + '&'
        }
     });
    console.log(bb);

    // 拼接app_key 
    //`string` 是模板字符串，ES2015新增的符号。
    // ${}里边 依旧是js，外边也是js
   let cc = `${bb}app_key=${"goaC95gnVNBtCNCv"}`
    console.log(cc)


    //MD5运算
    let dd = MD(cc);

    console.log(dd);

    // 转换大写
    let ee = dd.toUpperCase()
    console.log(ee);

    //添加sign到newdata中
    newData['sign'] = ee

    console.log(newData);
    //})




    })(this, null);





    //生成当前时间戳和随机字符串的函数
     function randomString(len) {
    len = len || 32;
    let timestamp = new Date().getTime();
    /****默认去掉了容易混淆的字符oOLl,9gq,Vv,Uu,I1****/
    let $chars = 'ABCDEFGHJKMNPQRSTWXYZabcdefhijkmnprstwxyz2345678';
    let maxPos = $chars.length;
    let randomStr = '';
   for (let i = 0; i < len; i++) {
     randomStr += $chars.charAt(Math.floor(Math.random() * maxPos));
   }
    let obj = {
    timestamp,
    randomStr
    }
    return obj;
    }
